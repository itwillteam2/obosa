//회원가입

pw varchar(20),

address varchar(20));

//공지사항
create table board(
num int primary key,
name varchar(20),
passwd varchar(20),
subject varchar(50),
content varchar(2000),
re_ref int,
re_lev int,
re_seq int,
readcount int,
date datetime,
ip varchar(20)
);
